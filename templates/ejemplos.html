Factorial de un número (versión iterativa)


      
   programa

   funciones:
   numero funcion factorial(numero n) {
      variables:
      numero resultado, contador;
      resultado = 1
      contador = 1

      mientras sucede (contador <= n) realiza {
         resultado = resultado * contador
         contador = contador + 1
      }
      
      devolver resultado
   }

   {
      imprimir ("1! = ", factorial(1))
      imprimir ("3! = ", factorial(3))
      imprimir ("5! = ", factorial(5))
      imprimir ("7! = ", factorial(7))
      imprimir ("10! = ", factorial(10))
   }

   // 0! = 1
   // 1! = 1
   // 2! = 2
   // 3! = 6
   // 4! = 24
   // 5! = 120
   // 6! = 720
   // 7! = 5040
   // 8! = 40320
   // 9! = 362880
   // 10! = 3628800

1! = 1
3! = 6
5! = 120
7! = 5040
10! = 3628800

Factorial de un número (versión recursiva)


      
   programa

   funciones:
   numero funcion factorial(numero n) {      
      si sucede (n <= 1) realiza {
         devolver 1
      } no sucede {
         devolver n * factorial(n - 1)
      }
   }

   {
      imprimir ("1! = ", factorial(1))
      imprimir ("3! = ", factorial(3))
      imprimir ("5! = ", factorial(5))

Fibonacci de un número (versión iterativa)


      
   programa

   funciones:
   numero funcion fibo(numero n) {
      variables:
      numero fibo, fiboAnterior, contador, temporal;

      fibo = 1
      fiboAnterior = 1

      si sucede (n <= 1) realiza {
         devolver n
      }

      contador = 2
      
      mientras sucede (contador < n) realiza {
         temporal = fibo
         fibo = fibo + fiboAnterior
         fiboAnterior = temporal
         contador = contador + 1
      }

      devolver fibo
   }

   {
      imprimir ("Fibo(1): ", fibo(1), "")
      imprimir ("Fibo(4): ", fibo(4), "")
      imprimir ("Fibo(10): ", fibo(10), "")
      imprimir ("Fibo(20): ", fibo(20), "")
   }

   // Fibo(0) = 0
   // Fibo(1) = 1
   // Fibo(2) = 1
   // Fibo(3) = 2
   // Fibo(4) = 3
   // Fibo(5) = 5
   // Fibo(10) = 55
   // Fibo(20) = 6765

Fibo(1): 1
Fibo(4): 3
Fibo(10): 55
Fibo(20): 6765

Ordenar una lista


      
   programa

   variables:
   lista de numero de 5 miListaDeNumeros;

   funciones:
   funcion ordenaLista(numero tamanio) {
      variables:
      numero i, j, temporal;
      i = 0

      mientras sucede (i < tamanio - 1) realiza {
         j = 0
         mientras sucede (j < tamanio - 1) realiza {
            si sucede (acceder (miListaDeNumeros, j) > acceder (miListaDeNumeros, j + 1)) realiza {
               temporal = acceder (miListaDeNumeros, j)
               agregar a (miListaDeNumeros, acceder (miListaDeNumeros, j + 1), j)
               agregar a (miListaDeNumeros, temporal, j + 1)
            }
            
            j = j + 1
         } 

         i = i + 1
      }
   }

   {
      agregar a (miListaDeNumeros, 20, 0)
      agregar a (miListaDeNumeros, 15, 1)
      agregar a (miListaDeNumeros, 10, 2)
      agregar a (miListaDeNumeros, 5, 3)
      agregar a (miListaDeNumeros, 1, 4)

      imprimir ("miListaDeNumeros (desordenada) = ", miListaDeNumeros)
      ordenaLista(5)
      imprimir ("miListaDeNumeros (ordenada) = ", miListaDeNumeros)
   }

miListaDeNumeros (desordenada) = [20, 15, 10, 5, 1]
miListaDeNumeros (ordenada) = [1, 5, 10, 15, 20]

Buscar un elemento en una lista


      
   programa

   variables:
   numero miNumeroABuscar;
   lista de numero de 5 miListaDeNumeros;

   funciones:
   binario funcion buscaEnListaElNumero(numero numeroABuscar, numero tamanio) {
      variables:
      numero i;
      i = 0

      mientras sucede (i < tamanio - 1) realiza {
         si sucede (acceder (miListaDeNumeros, i) == numeroABuscar) realiza {
            devolver verdadero
         }
         
         i = i + 1
      }

      devolver falso
   }

   {
      agregar a (miListaDeNumeros, 20, 0)
      agregar a (miListaDeNumeros, 15, 1)
      agregar a (miListaDeNumeros, 10, 2)
      agregar a (miListaDeNumeros, 5, 3)
      agregar a (miListaDeNumeros, 1, 4)
      miNumeroABuscar = 100

      imprimir ("miListaDeNumeros = ", miListaDeNumeros)
      
      si sucede (buscaEnListaElNumero(miNumeroABuscar, 5)) realiza {
         imprimir (miNumeroABuscar, " si está en ", miListaDeNumeros)
      } no sucede {
         imprimir (miNumeroABuscar, " no está en ", miListaDeNumeros)
      }

      miNumeroABuscar = 10

      si sucede (buscaEnListaElNumero(miNumeroABuscar, 5)) realiza {
         imprimir (miNumeroABuscar, " si está en ", miListaDeNumeros)
      } no sucede {
         imprimir (miNumeroABuscar, " no está en ", miListaDeNumeros)
      }
   }

miListaDeNumeros = [20, 15, 10, 5, 1]
100 no está en [20, 15, 10, 5, 1]
10 si está en [20, 15, 10, 5, 1]

Multiplicación de dos listas


      
   programa

   variables: 
   lista de numero de 3 miListaDeNumeros1, miListaDeNumeros2;

   funciones:
   lista de numero de 3 funcion multiplicaDosListas(numero tamanio) {
      variables:
      numero i;
      lista de numero de 3 miListaDeResultados;
      i = 0

      mientras sucede (i < tamanio) realiza {
         agregar a (miListaDeResultados, acceder (miListaDeNumeros1, i) * acceder (miListaDeNumeros2, i), i)
         i = i + 1
      }

      devolver miListaDeResultados
   }

   {
      agregar a (miListaDeNumeros1, 1, 0)
      agregar a (miListaDeNumeros1, 2, 1)
      agregar a (miListaDeNumeros1, 3, 2)
      agregar a (miListaDeNumeros2, 10, 0)
      agregar a (miListaDeNumeros2, 20, 1)
      agregar a (miListaDeNumeros2, 30, 2)
   
      imprimir ("miListaDeNumeros1 = ", miListaDeNumeros1)
      imprimir ("miListaDeNumeros2 = ", miListaDeNumeros2)
      imprimir ("miListaDeResultados = ", multiplicaDosListas(3))
      
   }

miListaDeNumeros1 = [1, 2, 3]
miListaDeNumeros2 = [10, 20, 30]
miListaDeResultados = [10, 40, 90]
