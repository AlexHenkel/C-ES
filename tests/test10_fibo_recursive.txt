// Test fibonacci recursive
programa

funciones:
numero funcion fibo(numero n) {
   variables:
   numero n1, n2;

   si sucede (n == 0 o n == 1) realiza {
      devolver n
   } no sucede {
      n1 = fibo(n - 1)
      n2 = fibo(n - 2)
      devolver n1 + n2
   }
}

{
   imprimir ("Fibo(1): ", fibo(1), "")
   imprimir ("Fibo(4): ", fibo(4), "")
   imprimir ("Fibo(10): ", fibo(10), "")
   imprimir ("Fibo(20): ", fibo(20), "")
}
