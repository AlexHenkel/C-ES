numero funcion factorial(numero n) {
   variables:
   numero temporal;

   si sucede (n <= 1) realiza {
      devolver 1
   } no sucede {
      temporal = factorial(n - 1)
      devolver n * temporal
   }
}

{
   imprimir ("1! = ", factorial(1))
   imprimir ("3! = ", factorial(3))
   imprimir ("5! = ", factorial(5))
   imprimir ("7! = ", factorial(7))
   imprimir ("10! = ", factorial(10))
}
