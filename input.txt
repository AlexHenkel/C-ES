funciones

numero funcion fibo(numero n) {
    variables: numero n1, n2;
    si sucede (n == 0 o n == 1) realiza {
        devolver n
    } no sucede {
        n1 = fibo(n - 1)
        n2 = fibo(n - 2)
        devolver n1 + n2
    }
}

{
    imprimir(fibo(0))
    imprimir(fibo(1))
    imprimir(fibo(2))
    imprimir(fibo(3))
    imprimir(fibo(4))
    imprimir(fibo(5))
    imprimir(fibo(10))
    imprimir(fibo(20))
}
